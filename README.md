# Введення даних
x = float(input("Введіть суму грошей Петрика (грн): "))
a = float(input("Введіть ціну одного квитка в кіно (грн): "))
b = float(input("Введіть ціну однієї порції морозива (грн): "))

# Обчислення загальної вартості
S = 2 * a + 2 * b

# Перевірка гіпотези
if x >= S:
    print("Грошей вистачить на 2 квитки і 2 порції морозива.")
else:
    print("Грошей не вистачить.")
